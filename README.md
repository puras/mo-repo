Mo Maven Repo
----

In your maven pom.xml:

    // GitHub
    <repository>
        <id>mo-repo</id>
        <name>Mo Maven Repo</name>
        <url>https://raw.githubusercontent.com/puras/mo-repo/master/repository/</url>
    </repository>

    // Gitee
    <repository>
        <id>mo-repo</id>
        <name>Mo Maven Repo</name>
        <url>https://gitee.com/puras/mo-repo/raw/master/repository/</url>
    </repository>
